P1
Create Your ADO Personal Access Token
~30 min

A Personal Access Token (PAT) is like a password that lets your app read data from ADO. You create it once and store it securely in your backend.

🚨
Security Rule #1 — Never paste your PAT into your React code or push it to GitHub. It goes into a backend environment variable only. Treat it like a bank password.
Step-by-Step: Create the PAT
1
Navigate to Personal Access Tokens
Open your browser → go to https://dev.azure.com/[YOUR-ORG] → click your profile picture (top right) → select Personal Access Tokens
2
Create a New Token
Click + New Token (blue button top right) → fill in:
Name: ADO-Daily-Agent  |  Organization: your org  |  Expiration: 90 days  |  Scopes: Custom defined
3
Set Minimum Required Permissions
Under Scopes, tick ONLY these:
✅ Work Items → Read
✅ Project and Team → Read
✅ Member Entitlement Management → Read (for leave/capacity)
4
Copy and Save the Token
Click Create — the token string appears exactly once. Copy it immediately and paste it into a password manager. You cannot retrieve it again after closing this screen.
5
Test Your PAT in the Browser
Paste this URL into a new browser tab — you should see JSON data come back:
https://dev.azure.com/YOUR-ORG/YOUR-PROJECT/_apis/wit/workitems?ids=1&api-version=7.1
If prompted for credentials, enter anything as username and your PAT as password.

2B — GetSprintData/index.js — Fetches Live ADO Data
GetSprintData/index.js
JavaScript 
const fetch = require('node-fetch');

// Read secrets from environment variables — never hardcode these
const ADO_ORG      = process.env.ADO_ORG;
const ADO_PROJECT  = process.env.ADO_PROJECT;
const ADO_PAT      = process.env.ADO_PAT;

// Base64 encode the PAT for ADO authentication
const adoAuth = 'Basic ' + Buffer.from(':' + ADO_PAT).toString('base64');
const adoBase = `https://dev.azure.com/${ADO_ORG}/${ADO_PROJECT}/_apis`;

// ── Fetch current sprint info ──────────────────────────────────────────
async function getCurrentSprint() {
  const url = `${adoBase}/work/teamsettings/iterations?$timeframe=current&api-version=7.1`;
  const res  = await fetch(url, { headers: { Authorization: adoAuth } });
  const data = await res.json();
  return data.value?.[0] || null;
}

// ── Fetch work items for the sprint ───────────────────────────────────
async function getWorkItems(iterationPath) {
  // Step 1: WIQL query to get IDs
  const wiql = { query: `SELECT [System.Id] FROM WorkItems
    WHERE [System.IterationPath] = '${iterationPath}'
    AND [System.State] NOT IN ('Removed')
    ORDER BY [System.WorkItemType]` };
  const qRes  = await fetch(`${adoBase}/wit/wiql?api-version=7.1`, {
    method: 'POST',
    headers: { Authorization: adoAuth, 'Content-Type': 'application/json' },
    body: JSON.stringify(wiql)
  });
  const qData = await qRes.json();
  const ids   = qData.workItems?..map(i => i.id).slice(0, 50).join(',');
  if (!ids) return [];

  // Step 2: Fetch full details for those IDs
  const fields = [
    'System.Id', 'System.Title', 'System.WorkItemType', 'System.State',
    'System.AssignedTo', 'System.IterationPath',
    'Microsoft.VSTS.Common.Priority', 'Microsoft.VSTS.Common.Severity',
    'Microsoft.VSTS.Scheduling.StoryPoints',
    'Microsoft.VSTS.Scheduling.TargetDate',
  ].join(',');
  const dRes  = await fetch(`${adoBase}/wit/workitems?ids=${ids}&fields=${fields}&api-version=7.1`,
    { headers: { Authorization: adoAuth } });
  const dData = await dRes.json();
  return dData.value || [];
}

// ── Fetch team capacity / leave plans ─────────────────────────────────
async function getCapacity(iterationId) {
  const url = `${adoBase}/work/teamsettings/iterations/${iterationId}/capacities?api-version=7.1`;
  const res  = await fetch(url, { headers: { Authorization: adoAuth } });
  const data = await res.json();
  return data.value || [];
}

// ── Main handler ──────────────────────────────────────────────────────
module.exports = async function(context, req) {
  try {
    const sprint   = await getCurrentSprint();
    if (!sprint) { context.res = { status: 404, body: { error: 'No active sprint' } }; return; }

    const [rawItems, capacity] = await Promise.all([
      getWorkItems(sprint.path),
      getCapacity(sprint.id)
    ]);

    const workItems = rawItems.map(item => ({
      id:       item.fields['System.Id'],
      title:    item.fields['System.Title'],
      type:     item.fields['System.WorkItemType'],
      state:    item.fields['System.State'],
      assignee: item.fields['System.AssignedTo']?.displayName || 'Unassigned',
      points:   item.fields['Microsoft.VSTS.Scheduling.StoryPoints'] || 0,
      eta:      item.fields['Microsoft.VSTS.Scheduling.TargetDate']?.split('T')[0] || null,
      priority: item.fields['Microsoft.VSTS.Common.Priority'] || '4 - Low',
    }));

    const leavePlans = capacity
      .filter(c => c.daysOff?.length > 0)
      .flatMap(c => c.daysOff.map(d => ({
        person: c.teamMember.displayName,
        from: d.start.split('T')[0],
        to:   d.end.split('T')[0],
        type: 'Leave'
      })));

    context.res = {
      status: 200,
      headers: { 'Content-Type': 'application/json', 'Access-Control-Allow-Origin': '*' },
      body: JSON.stringify({ sprint, workItems, leavePlans })
    };
  } catch (err) {
    context.res = { status: 500, body: { error: err.message } };
  }
};
2C — GenerateSummary/index.js — Secure Claude API Proxy
GenerateSummary/index.js
JavaScript 
const fetch = require('node-fetch');

module.exports = async function(context, req) {
  const { messages, systemPrompt } = req.body;

  const res = await fetch('https://api.anthropic.com/v1/messages', {
    method: 'POST',
    headers: {
      'Content-Type':      'application/json',
      'x-api-key':         process.env.ANTHROPIC_API_KEY,
      'anthropic-version': '2023-06-01'
    },
    body: JSON.stringify({
      model:      'claude-sonnet-4-20250514',
      max_tokens: 1000,
      system:     systemPrompt,
      messages:   messages
    })
  });

  const data = await res.json();
  context.res = {
    status: 200,
    headers: { 'Content-Type': 'application/json', 'Access-Control-Allow-Origin': '*' },
    body: JSON.stringify({ reply: data.content?.[0]?.text || '' })
  };
};

3A — Add Live Data Fetching

At the top of your component, replace the hardcoded WORK_ITEMS, LEAVE_PLANS and SPRINT constants with this:

ado-daily-summary-agent.jsx — replace mock data section 
// Point to your backend (local while testing, Azure URL after deploying)
const BACKEND_URL = 'http://localhost:7071/api'; // update after deploy

// Inside your component, replace useState initial values:
const [sprintData,  setSprintData]  = useState(null);
const [workItems,   setWorkItems]   = useState([]);
const [leavePlans,  setLeavePlans]  = useState([]);
const [dataLoading, setDataLoading] = useState(true);
const [dataError,   setDataError]   = useState(null);

// Add this useEffect to load data on mount + auto-refresh every 5 minutes
useEffect(() => {
  async function loadData() {
    try {
      setDataLoading(true);
      const res  = await fetch(`${BACKEND_URL}/GetSprintData`);
      const data = await res.json();
      setSprintData(data.sprint);
      setWorkItems(data.workItems);
      setLeavePlans(data.leavePlans);
    } catch (err) {
      setDataError('Could not load ADO data: ' + err.message);
    } finally {
      setDataLoading(false);
    }
  }
  loadData();
  const interval = setInterval(loadData, 5 * 60 * 1000); // refresh every 5 min
  return () => clearInterval(interval);
}, []);

Deploy to Azure

4A — Deploy the Backend (Azure Function App)
Terminal — run these commands in order 
# 1. Create a resource group
az group create --name ado-agent-rg --location eastus

# 2. Create a storage account (required by Azure Functions)
az storage account create \
  --name adoagentstorage \
  --location eastus \
  --resource-group ado-agent-rg \
  --sku Standard_LRS

# 3. Create the Function App
az functionapp create \
  --resource-group ado-agent-rg \
  --consumption-plan-location eastus \
  --runtime node --runtime-version 18 \
  --functions-version 4 \
  --name ado-daily-agent-api \
  --storage-account adoagentstorage

# 4. Upload your secrets as environment variables (NOT in code)
az functionapp config appsettings set \
  --name ado-daily-agent-api \
  --resource-group ado-agent-rg \
  --settings \
    ADO_ORG="your-org" \
    ADO_PROJECT="your-project" \
    ADO_PAT="your-pat-here" \
    ANTHROPIC_API_KEY="sk-ant-your-key"

# 5. Deploy your code
func azure functionapp publish ado-daily-agent-api

Your backend is now live at: https://ado-daily-agent-api.azurewebsites.net/api/GetSprintData

4C — Daily Email Summary (Scheduled Azure Function)

Add a timer-triggered function that sends the daily summary email at 8am every weekday.

DailySummaryEmail/function.json 
{
  "bindings": [{
    "name":      "myTimer",
    "type":      "timerTrigger",
    "direction": "in",
    "schedule":  "0 0 8 * * 1-5"
  }]
}
// 0 0 8 * * 1-5  = 8:00am UTC, Monday–Friday
// For IST (UTC+5:30): use "0 30 2 * * 1-5"
// For EST (UTC-5):    use "0 0 13 * * 1-5"
DailySummaryEmail/index.js 
const fetch       = require('node-fetch');
const nodemailer = require('nodemailer');

module.exports = async function(context, myTimer) {
  // 1. Fetch live ADO data from your own endpoint
  const adoRes  = await fetch('https://ado-daily-agent-api.azurewebsites.net/api/GetSprintData');
  const adoData = await adoRes.json();

  // 2. Ask Claude to generate the daily summary
  const aiRes = await fetch('https://api.anthropic.com/v1/messages', {
    method:  'POST',
    headers: {
      'Content-Type':      'application/json',
      'x-api-key':         process.env.ANTHROPIC_API_KEY,
      'anthropic-version': '2023-06-01'
    },
    body: JSON.stringify({
      model:      'claude-sonnet-4-20250514',
      max_tokens: 1000,
      messages:   [{ role: 'user', content: 'Generate today daily sprint summary with all risks and leave impact' }],
      system:     buildSystemPrompt(adoData)  // reuse your existing function
    })
  });
  const aiData  = await aiRes.json();
  const summary = aiData.content?.[0]?.text || 'Could not generate summary';

  // 3. Send the email via SMTP (Office 365 / Gmail)
  const transporter = nodemailer.createTransport({
    host: 'smtp.office365.com',  // swap with smtp.gmail.com for Gmail
    port: 587,
    auth: { user: process.env.EMAIL_FROM, pass: process.env.EMAIL_PASSWORD }
  });
  await transporter.sendMail({
    from:    process.env.EMAIL_FROM,
    to:      process.env.EMAIL_RECIPIENTS,
    subject: `Daily Sprint Summary — ${new Date().toDateString()}`,
    html:    `<pre style="font-family:sans-serif;line-height:1.6">${summary}</pre>`
  });
  context.log('Daily summary sent ✓');
};
